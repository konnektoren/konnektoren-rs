13/// Represents challenge-level commands that can be executed on the game state.
14#[derive(Debug, Clone, PartialEq)]
15pub enum ChallengeCommand {
16    Start(Challenge),
17    /// Command to move to the next task within a challenge.
18    NextTask,
19    /// Command to move to the previous task within a challenge.
20    PreviousTask,
21    /// Command to solve a multiple choice option.
22    SolveOption(usize),
23    /// Command to finish the challenge with a custom result.
24    Finish(Option<ChallengeResult>),
25}
26
27impl CommandTrait for ChallengeCommand {
28    /// Executes the challenge command on the given game state.
29    ///
30    /// # Arguments
31    ///
32    /// * `state` - A mutable reference to the current game state.
33    ///
34    /// # Returns
35    ///
36    /// A `Result` indicating success or containing an error if the command execution failed.
37    fn execute(&self, state: &mut GameState) -> Result<()> {
38        match self {
39            ChallengeCommand::Start(challenge_type) => Self::start_challenge(state, challenge_type),
40            ChallengeCommand::NextTask => Self::next_task(state),
41            ChallengeCommand::PreviousTask => Self::previous_task(state),
42            ChallengeCommand::SolveOption(option_index) => Self::solve_option(state, *option_index),
43            ChallengeCommand::Finish(result) => Self::finish_challenge(state, result),
44        }
45    }
46
47    /// Gets the type of the command.
48    fn get_type(&self) -> CommandType {
49        CommandType::Challenge
50    }
51}
52
53impl ChallengeCommand {
54    /// Starts a new challenge with the given challenge configuration.
55    fn start_challenge(state: &mut GameState, challenge: &Challenge) -> Result<()> {
56        let mut challenge = challenge.clone();
57        challenge.start();
58        state.challenge = challenge;
59        state.current_task_index = 0;
60        Ok(())
61    }
62
63    /// Moves the game state to the next task within the current challenge.
64    ///
65    /// # Arguments
66    ///
67    /// * `state` - A mutable reference to the current game state.
68    ///
69    /// # Returns
70    ///
71    /// A `Result` indicating success or containing an error if there are no more tasks.
72    fn next_task(state: &mut GameState) -> Result<()> {
73        let current_game_path: &GamePath = state
74            .game
75            .game_paths
76            .get(state.current_game_path)
77            .expect("Invalid game path index");
78        let challenge_config = &current_game_path.challenges[state.current_challenge_index];
79        let max_questions = challenge_config.tasks.len();
80        if state.current_task_index >= max_questions - 1 {
81            return Err(anyhow!("No more tasks"));
82        }
83        state.current_task_index += 1;
84        Ok(())
85    }
86
87    /// Moves the game state to the previous task within the current challenge.
88    ///
89    /// # Arguments
90    ///
91    /// * `state` - A mutable reference to the current game state.
92    ///
93    /// # Returns
94    ///
95    /// A `Result` indicating success or containing an error if there are no previous tasks.
96    fn previous_task(state: &mut GameState) -> Result<()> {
97        if state.current_task_index == 0 {
98            return Err(anyhow!("No previous tasks"));
99        }
100        state.current_task_index -= 1;
101        Ok(())
102    }
103
104    /// Solves the current task with the selected option and moves to the next task.
105    ///
106    /// # Arguments
107    ///
108    /// * `state` - A mutable reference to the current game state.
109    /// * `option_index` - The index of the selected option.
110    ///
111    /// # Returns
112    ///
113    /// A `Result` indicating success or containing an error if the solution is invalid.
114    fn solve_option(state: &mut GameState, option_index: usize) -> Result<()> {
115        let challenge_input = match state.challenge.challenge_type {
116            ChallengeType::MultipleChoice(ref dataset) => {
117                let option = match dataset.options.get(option_index) {
118                    Some(option) => option,
119                    None => {
120                        return Err(anyhow!("Invalid option id: {}", option_index));
121                    }
122                };
123                ChallengeInput::MultipleChoice(MultipleChoiceOption {
124                    id: option.id,
125                    name: option.name.clone(),
126                })
127            }
128            _ => {
129                return Err(anyhow!("Invalid challenge type"));
130            }
131        };
132        state.challenge.solve(challenge_input)?;
133
134        Self::next_task(state).unwrap_or_default();
135
136        Ok(())
137    }
138
139    /// Finishes the current challenge with a custom result.
140    fn finish_challenge(
141        state: &mut GameState,
142        custom_result: &Option<ChallengeResult>,
143    ) -> Result<()> {
144        state.challenge.update_end_time();
145        // Logic to handle finishing the challenge
146        if let Some(result) = custom_result {
147            state.challenge.challenge_result = result.clone();
148        }
149        // Additional logic to mark the challenge as finished
150        Ok(())
151    }
152}
