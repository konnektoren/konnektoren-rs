1use schemars::JsonSchema;
2use serde::{Deserialize, Serialize};
3
4#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, JsonSchema)]
5pub struct ContextualChoice {
6    /// Unique identifier for the challenge
7    pub id: String,
8    /// Display name of the challenge
9    pub name: String,
10    /// Description of the challenge
11    pub description: String,
12    /// List of contextual items
13    pub items: Vec<ContextItem>,
14}
15
16impl Default for ContextualChoice {
17    fn default() -> Self {
18        let data = include_str!("../../../assets/contextual_choice_default.yml");
19        serde_yaml::from_str(data).unwrap()
20    }
21}
22
23#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, Default, JsonSchema)]
24pub struct ContextItem {
25    /// Template string with placeholders like {0}, {1}
26    pub template: String,
27    /// Available choices for each placeholder
28    pub choices: Vec<Choice>,
29}
30
31#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, Default, JsonSchema)]
32pub struct Choice {
33    /// Choice identifier
34    pub id: usize,
35    /// Available options for this choice
36    pub options: Vec<String>,
37    /// The correct answer
38    pub correct_answer: String,
39}
40
41#[derive(Debug, Serialize, Deserialize, Clone, PartialEq, Default, JsonSchema)]
42pub struct ContextItemChoiceAnswers {
43    /// Selected option IDs for each choice
44    pub ids: Vec<usize>,
45}

52    #[test]
53    fn test_contextual_choice_deserialization() {
54        let yaml = r#"
55        id: "konjunktiv-2"
56        name: "Konjunktiv II Exercise"
57        description: "Fill in the gaps with the correct Konjunktiv II forms."
58        items:
59          - template: "Wenn ich reich {0}, {1} ich um die Welt reisen."
60            choices:
61              - id: 0
62                options:
63                  - "wäre"
64                  - "würde sein"
65                  - "sei"
66                correct_answer: "wäre"
67              - id: 1
68                options:
69                  - "würde"
70                  - "werde"
71                  - "wurde"
72                correct_answer: "würde"
73          - template: "Er {0} gerne Deutsch, wenn er mehr Zeit {1}."
74            choices:
75              - id: 0
76                options:
77                  - "lernte"
78                  - "würde lernen"
79                  - "lerne"
80                correct_answer: "würde lernen"
81              - id: 1
82                options:
83                  - "hätte"
84                  - "habe"
85                  - "würde haben"
86                correct_answer: "hätte"
87        "#;
88
89        let contextual_choice: ContextualChoice = serde_yaml::from_str(yaml).unwrap();
90
91        assert_eq!(contextual_choice.id, "konjunktiv-2");
92        assert_eq!(contextual_choice.name, "Konjunktiv II Exercise");
93        assert_eq!(
94            contextual_choice.description,
95            "Fill in the gaps with the correct Konjunktiv II forms."
96        );
97        assert_eq!(contextual_choice.items.len(), 2);
98
99        let first_item = &contextual_choice.items[0];
100        assert_eq!(
101            first_item.template,
102            "Wenn ich reich {0}, {1} ich um die Welt reisen."
103        );
104        assert_eq!(first_item.choices.len(), 2);
105        assert_eq!(first_item.choices[0].id, 0);
106        assert_eq!(
107            first_item.choices[0].options,
108            vec!["wäre", "würde sein", "sei"]
109        );
110        assert_eq!(first_item.choices[0].correct_answer, "wäre");
111
112        let second_item = &contextual_choice.items[1];
113        assert_eq!(
114            second_item.template,
115            "Er {0} gerne Deutsch, wenn er mehr Zeit {1}."
116        );
117        assert_eq!(second_item.choices.len(), 2);
118        assert_eq!(second_item.choices[1].id, 1);
119        assert_eq!(
120            second_item.choices[1].options,
121            vec!["hätte", "habe", "würde haben"]
122        );
123        assert_eq!(second_item.choices[1].correct_answer, "hätte");
124    }
