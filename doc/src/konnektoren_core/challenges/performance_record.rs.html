6pub type ChallengeTimeMilliseconds = u64;
7pub type ChallengeId = String;
8pub type ChallengePercentage = u8;
9pub type ChallengePerformance = (ChallengeId, ChallengePercentage, ChallengeTimeMilliseconds);
10
11#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Eq, Hash, Default)]
12pub struct PerformanceRecord {
13    pub game_path_id: String,
14    pub profile_name: String,
15    pub challenges_performance: Vec<ChallengePerformance>,
16    pub total_challenges: usize,
17    pub performance_percentage: u8,
18    pub date: DateTime<Utc>,
19}
20
21impl PerformanceRecord {
22    pub fn new(
23        game_path_id: String,
24        user_profile_name: String,
25        challenges_performance: Vec<ChallengePerformance>,
26        total_challenges: usize,
27        date: DateTime<Utc>,
28    ) -> Self {
29        let solved_challenges = challenges_performance.len();
30        let performance_percentage =
31            ((solved_challenges as f64 / total_challenges as f64) * 100.0) as u8;
32        PerformanceRecord {
33            game_path_id,
34            profile_name: user_profile_name,
35            challenges_performance,
36            total_challenges,
37            performance_percentage,
38            date,
39        }
40    }
41
42    pub fn new_from_history(
43        game_path_id: String,
44        user_profile_name: String,
45        total_challenges: usize,
46        challenge_history: ChallengeHistory,
47    ) -> Self {
48        let mut best_performance: HashMap<ChallengeId, ChallengePerformance> = HashMap::new();
49
50        for challenge in challenge_history.challenges {
51            let id = challenge.challenge_config.id.clone();
52            let performance = challenge.performance(&challenge.challenge_result) as u8;
53            let elapsed_time = challenge
54                .elapsed_time()
55                .map(|d| d.num_milliseconds() as ChallengeTimeMilliseconds)
56                .unwrap_or(0);
57
58            let current_performance = best_performance.get(&id).map_or(0, |(_, p, _)| *p);
59
60            best_performance
61                .entry(id.clone())
62                .and_modify(|(_id, p, t)| {
63                    if performance > *p {
64                        *p = performance;
65                        *t = elapsed_time;
66                    } else if elapsed_time < *t && performance == current_performance {
67                        *t = elapsed_time;
68                    }
69                })
70                .or_insert((id, performance, elapsed_time));
71        }
72
73        let mut challenges_performance: Vec<ChallengePerformance> =
74            best_performance.values().cloned().collect();
75
76        // Sort by performance in descending order (highest performance first)
77        challenges_performance.sort_by(|a, b| b.1.cmp(&a.1));
78
79        // Sort by elapsed time in ascending order (lowest time first)
80        challenges_performance.sort_by(|a, b| a.2.cmp(&b.2));
81
82        let solved_challenges = challenges_performance.len();
83        let performance_percentage =
84            ((solved_challenges as f64 / total_challenges as f64) * 100.0) as u8;
85
86        PerformanceRecord {
87            game_path_id,
88            profile_name: user_profile_name,
89            challenges_performance,
90            total_challenges,
91            performance_percentage,
92            date: Utc::now(),
93        }
94    }
95}

97impl Timed for PerformanceRecord {
98    fn start(&mut self) {}
99    fn update_end_time(&mut self) {}
100    fn elapsed_time(&self) -> Option<Duration> {
101        let mut elapsed_time = Duration::zero();
102        for (_, _, time) in &self.challenges_performance {
103            elapsed_time += Duration::milliseconds(*time as i64);
104        }
105        Some(elapsed_time)
106    }
107    fn start_time(&self) -> Option<DateTime<Utc>> {
108        Some(self.date)
109    }
110    fn end_time(&self) -> Option<DateTime<Utc>> {
111        let end_time = self.date + self.elapsed_time().unwrap_or_default();
112        Some(end_time)
113    }
114}

116impl PartialOrd for PerformanceRecord {
117    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
118        match other
119            .performance_percentage
120            .cmp(&self.performance_percentage)
121        {
122            std::cmp::Ordering::Equal => Some(
123                self.elapsed_time()
124                    .and_then(|self_elapsed| {
125                        other
126                            .elapsed_time()
127                            .map(|other_elapsed| self_elapsed.cmp(&other_elapsed))
128                    })
129                    .map(|time_ord| time_ord.then_with(|| other.date.cmp(&self.date)))
130                    .unwrap_or(std::cmp::Ordering::Equal),
131            ),
132            other => Some(other),
133        }
134    }
135}
136
137impl Ord for PerformanceRecord {
138    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
139        self.partial_cmp(other).unwrap()
140    }
141}
