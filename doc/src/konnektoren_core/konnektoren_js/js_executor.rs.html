1use super::KONNEKTOREN_NAMESPACE;
2use wasm_bindgen::JsCast;
3use wasm_bindgen::JsValue;
4use web_sys::Window;
5
6pub struct JsExecutor<'a> {
7    window: &'a Window,
8}
9
10impl<'a> JsExecutor<'a> {
11    pub fn new(window: &'a Window) -> Self {
12        Self { window }
13    }
14
15    pub fn execute(&self, js_code: &str) {
16        if let Err(err) = js_sys::eval(js_code) {
17            log::error!("JavaScript execution failed: {:?}", err);
18        }
19    }
20
21    pub fn get_or_create_konnektoren_object(&self) -> JsValue {
22        let global = JsValue::from(self.window.clone());
23        if js_sys::Reflect::has(&global, &JsValue::from_str(KONNEKTOREN_NAMESPACE)).unwrap_or(false)
24        {
25            js_sys::Reflect::get(&global, &JsValue::from_str(KONNEKTOREN_NAMESPACE)).unwrap()
26        } else {
27            let obj = js_sys::Object::new();
28            js_sys::Reflect::set(&global, &JsValue::from_str(KONNEKTOREN_NAMESPACE), &obj).unwrap();
29            obj.into()
30        }
31    }
32
33    pub fn is_konnektoren_object_empty(&self, konnektoren_obj: &JsValue) -> bool {
34        js_sys::Object::keys(konnektoren_obj.unchecked_ref::<js_sys::Object>()).length() == 0
35    }
36
37    pub fn create_konnektoren_object(&self, konnektoren_obj: &JsValue) {
38        let version = env!("CARGO_PKG_VERSION");
39        js_sys::Reflect::set(
40            konnektoren_obj,
41            &JsValue::from_str("version"),
42            &JsValue::from_str(version),
43        )
44        .unwrap();
45    }
46}
