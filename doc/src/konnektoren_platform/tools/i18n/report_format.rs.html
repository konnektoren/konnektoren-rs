6    fn format(&self, report: &I18nReport) -> Result<String, I18nReportError>;
7}

13impl I18nReportFormatter for I18nYamlFormatter {
14    fn format(&self, report: &I18nReport) -> Result<String, I18nReportError> {
15        use std::collections::BTreeMap;
16        let mut i18n_map: BTreeMap<String, BTreeMap<String, String>> = BTreeMap::new();
17        for (lang, missing_keys) in &report.missing_translations {
18            for key in missing_keys {
19                i18n_map
20                    .entry(key.clone())
21                    .or_default()
22                    .insert(lang.clone(), "".to_string());
23            }
24        }
25        let mut yaml = String::from("i18n:\n");
26        for (key, langs) in i18n_map {
27            yaml.push_str(&format!("  \"{}\":\n", key));
28            for (lang, _) in langs {
29                yaml.push_str(&format!("    \"{}\": \"\"\n", lang));
30            }
31        }
32        Ok(yaml)
33    }
34}

36impl I18nReportFormatter for I18nJsonFormatter {
37    fn format(&self, report: &I18nReport) -> Result<String, I18nReportError> {
38        let mut map = serde_json::Map::new();
39        for (lang, missing_keys) in &report.missing_translations {
40            let mut lang_map = serde_json::Map::new();
41            for key in missing_keys {
42                lang_map.insert(key.clone(), serde_json::json!(""));
43            }
44            map.insert(lang.clone(), serde_json::json!(lang_map));
45        }
46        serde_json::to_string_pretty(&serde_json::json!(map)).map_err(I18nReportError::JsonError)
47    }
48}

50impl I18nReportFormatter for I18nHumanFormatter {
51    fn format(&self, report: &I18nReport) -> Result<String, I18nReportError> {
52        use std::fmt::Write;
53        let mut out = String::new();
54
55        writeln!(out, "\nTranslation Report").map_err(I18nReportError::FmtError)?;
56        writeln!(out, "=================").map_err(I18nReportError::FmtError)?;
57        writeln!(out, "Source keys found: {}", report.source_keys.len())
58            .map_err(I18nReportError::FmtError)?;
59
60        writeln!(out, "\nLanguage Statistics:").map_err(I18nReportError::FmtError)?;
61        writeln!(out, "-------------------").map_err(I18nReportError::FmtError)?;
62        for lang in crate::i18n::Language::builtin() {
63            if let Some(stats) = report.language_stats.get(lang.code()) {
64                writeln!(
65                    out,
66                    "{} ({}): {}/{} keys ({:.1}% coverage)",
67                    lang.native_name(),
68                    lang.code(),
69                    stats.total_keys - stats.missing_keys,
70                    stats.total_keys,
71                    stats.coverage_percentage
72                )
73                .map_err(I18nReportError::FmtError)?;
74            }
75        }
76
77        if !report.missing_translations.is_empty() {
78            writeln!(out, "\nMissing Translations:").map_err(I18nReportError::FmtError)?;
79            writeln!(out, "-------------------").map_err(I18nReportError::FmtError)?;
80            for lang in crate::i18n::Language::builtin() {
81                if let Some(missing) = report.missing_translations.get(lang.code()) {
82                    if !missing.is_empty() {
83                        writeln!(
84                            out,
85                            "{} ({}) - {} missing:",
86                            lang.native_name(),
87                            lang.code(),
88                            missing.len()
89                        )
90                        .map_err(I18nReportError::FmtError)?;
91                        for key in missing {
92                            if let Some(en_trans) = report
93                                .translations
94                                .get("en")
95                                .and_then(|t| t.get(key))
96                                .and_then(|v| v.as_str())
97                            {
98                                writeln!(out, "  - {}: \"{}\"", key, en_trans)
99                                    .map_err(I18nReportError::FmtError)?;
100                            } else {
101                                writeln!(out, "  - {}", key).map_err(I18nReportError::FmtError)?;
102                            }
103                        }
104                    }
105                }
106            }
107        }
108
109        if !report.unused_translations.is_empty() {
110            writeln!(out, "\nUnused Translations:").map_err(I18nReportError::FmtError)?;
111            writeln!(out, "-------------------").map_err(I18nReportError::FmtError)?;
112            for key in &report.unused_translations {
113                writeln!(out, "  - {}", key).map_err(I18nReportError::FmtError)?;
114                for lang in crate::i18n::Language::builtin() {
115                    if let Some(trans) = report
116                        .translations
117                        .get(lang.code())
118                        .and_then(|t| t.get(key))
119                        .and_then(|v| v.as_str())
120                    {
121                        writeln!(
122                            out,
123                            "    {} ({}): \"{}\"",
124                            lang.native_name(),
125                            lang.code(),
126                            trans
127                        )
128                        .map_err(I18nReportError::FmtError)?;
129                    }
130                }
131            }
132        }
133
134        writeln!(out, "\nSummary:").map_err(I18nReportError::FmtError)?;
135        writeln!(out, "--------").map_err(I18nReportError::FmtError)?;
136        writeln!(out, "Total source keys: {}", report.source_keys.len())
137            .map_err(I18nReportError::FmtError)?;
138        writeln!(
139            out,
140            "Languages: {}",
141            crate::i18n::Language::builtin()
142                .iter()
143                .map(|l| l.code())
144                .collect::<Vec<_>>()
145                .join(", ")
146        )
147        .map_err(I18nReportError::FmtError)?;
148
149        let total_missing: usize = report.missing_translations.values().map(|v| v.len()).sum();
150        writeln!(out, "Total missing translations: {}", total_missing)
151            .map_err(I18nReportError::FmtError)?;
152
153        writeln!(
154            out,
155            "Overall status: {}",
156            if report.has_errors {
157                "❌ Missing translations"
158            } else {
159                "✅ All translations complete"
160            }
161        )
162        .map_err(I18nReportError::FmtError)?;
163
164        Ok(out)
165    }
166}
