28impl<T: rust_embed::RustEmbed> TranslationAsset for JsonTranslationAsset<T> {
29    fn load_translations(&self) -> HashMap<String, serde_json::Value> {
30        let mut translations: HashMap<String, serde_json::Map<String, serde_json::Value>> =
31            HashMap::new();
32
33        // Collect all files in the embedded folder
34        for file in T::iter() {
35            let filename = file.as_ref();
36
37            // Try to extract the language code from the filename
38            // Accepts: de.json, level_a1_de.json, wortschatz_a1_begruessung_de.json, etc.
39            if let Some(lang) = filename.strip_suffix(".json") {
40                // Try to match "de" or "level_a1_de"
41                let parts: Vec<&str> = lang.split('_').collect();
42                let lang_code = if parts.len() == 1 {
43                    // de
44                    parts[0]
45                } else {
46                    // level_a1_de -> de
47                    parts.last().unwrap()
48                };
49
50                // Only process if it's a known language code
51                if Language::builtin().iter().any(|l| l.code() == lang_code) {
52                    if let Some(content) = T::get(filename) {
53                        if let Ok(json) = serde_json::from_slice::<
54                            serde_json::Map<String, serde_json::Value>,
55                        >(&content.data)
56                        {
57                            translations
58                                .entry(lang_code.to_string())
59                                .and_modify(|existing| existing.extend(json.clone()))
60                                .or_insert(json);
61                        }
62                    }
63                }
64            }
65        }
66
67        // Convert to HashMap<String, Value>
68        translations
69            .into_iter()
70            .map(|(k, v)| (k, serde_json::Value::Object(v)))
71            .collect()
72    }
73}

75// Implementation for embedded YAML files
76pub struct YamlTranslationAsset<T: rust_embed::RustEmbed> {
77    filename: String,
78    _marker: std::marker::PhantomData<T>,
79}
80
81impl<T: rust_embed::RustEmbed> YamlTranslationAsset<T> {
82    pub fn new(filename: &str) -> Self {
83        Self {
84            filename: filename.to_string(),
85            _marker: std::marker::PhantomData,
86        }
87    }
88
89    fn load_yaml(&self) -> Option<HashMap<String, Value>> {
90        T::get(&self.filename).and_then(|file| {
91            String::from_utf8(file.data.to_vec())
92                .ok()
93                .and_then(|content| {
94                    serde_yaml::from_str::<serde_yaml::Value>(&content)
95                        .ok()
96                        .and_then(|yaml| {
97                            yaml.get("i18n").map(|i18n| {
98                                let mut translations = HashMap::new();
99                                if let Some(mapping) = i18n.as_mapping() {
100                                    for (key, translations_map) in mapping {
101                                        let key = key.as_str().unwrap_or_default();
102                                        if let Some(trans_map) = translations_map.as_mapping() {
103                                            for (lang, trans) in trans_map {
104                                                let lang = lang.as_str().unwrap_or_default();
105                                                let trans = trans.as_str().unwrap_or_default();
106
107                                                translations
108                                                    .entry(lang.to_string())
109                                                    .or_insert_with(|| json!({}))
110                                                    .as_object_mut()
111                                                    .unwrap()
112                                                    .insert(
113                                                        key.to_string(),
114                                                        Value::String(trans.to_string()),
115                                                    );
116                                            }
117                                        }
118                                    }
119                                }
120                                translations
121                            })
122                        })
123                })
124        })
125    }
126}

128impl<T: rust_embed::RustEmbed> TranslationAsset for YamlTranslationAsset<T> {
129    fn load_translations(&self) -> HashMap<String, Value> {
130        self.load_yaml().unwrap_or_default()
131    }

134// Combined asset loader
135pub struct CombinedTranslationAsset<T: rust_embed::RustEmbed> {
136    json_loader: JsonTranslationAsset<T>,
137    yaml_loader: YamlTranslationAsset<T>,
138}
139
140impl<T: rust_embed::RustEmbed> CombinedTranslationAsset<T> {
141    pub fn new(yaml_filename: &str) -> Self {
142        Self {
143            json_loader: JsonTranslationAsset::new(),
144            yaml_loader: YamlTranslationAsset::new(yaml_filename),
145        }
146    }
147}
148
149impl<T: rust_embed::RustEmbed> TranslationAsset for CombinedTranslationAsset<T> {
150    fn load_translations(&self) -> HashMap<String, Value> {
151        let mut translations = self.json_loader.load_translations();
152
153        if let Some(yaml_translations) = self.yaml_loader.load_yaml() {
154            for (lang, trans) in yaml_translations {
155                translations
156                    .entry(lang)
157                    .and_modify(|e| {
158                        if let Some(obj) = e.as_object_mut() {
159                            if let Some(new_obj) = trans.as_object() {
160                                obj.extend(new_obj.clone());
161                            }
162                        }
163                    })
164                    .or_insert(trans);
165            }
166        }
167
168        translations
169    }
170}
